  • Thay can be called by the main program repeatedly as per the requirement.

    2.Write a program to find out factorial of number using UDF.



  • C program:

    • #include<stdio.h>
    #include<conio.h>

    int fact(int a);
    void main()
    {
    int x,z;
    clrscr();
    printf("\nEnter any Number:");
    scanf("%d",&x);

    z=fact(x);

    printf("\nFactorial Of a Number is:%d",z);
    getch();
    }
    int fact(int a)
    {
    int f=1,i=1;
    while(i<=a)
    {
    f=f*i;
    i++;
    }
    return(f);
    }

    5.Write a program to find out Number is Palindrome or not using UDF.



  • C program:

    • #include<stdio.h>
    #include<conio.h>

    int palindrome(int n);
    void main()
    {
    int x;
    clrscr();
    printf("\nEnter any Number:");
    scanf("%d",&x);

    palindrome(x);
    getch();
    }
    int palindrome(int n)
    {
    int a,sum=0,b;
    b=n;
    while(n>0)
    {
    a=n%10;
    sum=sum*10+a;
    n=n/10;
    }
    if(b==sum)
    {
    printf("\nNumber is Palindrome");
    }
    else
    {
    printf("\nNumber is not Palindrome");
    }
    }

    printf("\nInput Number is Armstrong");
    }
    else
    {
    printf("\nInput Number is Not Armstrong");
    }
    }

    7.Write a program to find out Number is Prime or not using UDF.



  • C program:

    • #include<stdio.h>
    #include<conio.h>

    int prime(int n);
    void main()
    {
    int x;
    clrscr();
    printf("\nEnter any Number:");
    scanf("%d",&x);

    prime(x);
    getch();
    }
    int prime(int n)
    {
    int i,f=0;
    for(i=1;i<=n;i++)
    {
    if(n%i==0)
    {
    f++;
    }
    }
    if(f==2)
    {
    printf("\nNumber is Prime");
    }
    else
    {
    printf("\nNumber is not Prime");
    }
    }


  • Output:

    • Enter any Number:3

    Number is Prime
