1.Write a program to read and display n numbers using one dimensional array.



  • C program:

    • #include<stdio.h>
    #include<conio.h>

    void main()
    {
    int a[100],i,n;
    clrscr();
    printf("\nHow many values you want to enter:");
    scanf("%d",&n);
    for(i=0;i < n;i++)
    {
    printf("\nEnter Value of element:");
    scanf("%d",&a[i]);
    }
    for(i=0;i < n;i++)
    {
    printf("\nValue of element %d:",a[i]);
    }
    getch();
    }

    2.Write a program to enter N elements and find total and average of them.



  • C program:

    • #include<stdio.h>
    #include<conio.h>

    void main()
    {
    int a[100],i,n;
    float avg,total;
    clrscr();
    printf("\nHow many numbers you want to print:");
    scanf("%d",&n);
    for(i=0;i < n;i++)
    {
    printf("\nEnter Value of element: ");
    scanf("%d",&a[i]);
    }
    for(i=0;i < n;i++)
    {
    total=total+a[i];
    }
    printf("\nTotal=%.2f",total);
    avg=total/n;
    printf("\nAverage=%.2f",avg);
    getch();
    }


  • Output:
    • How many numbers you want to print:3

    Enter Value of element:2
    Enter Value of element:1
    Enter Value of element:3

    Total=6
    Average=2

    3.Write a program to enter N elements and find Minimum and Maximum Value from the array.



  • C program:

    • #include<stdio.h>
    #include<conio.h>

    void main()
    {
    int a[100],i,n,mini,max;
    clrscr();
    printf("\nEnter how many number you want to print:");
    scanf("%d",&n);
    for(i=0;i < n;i++)
    {
    printf("\nEnter Value of element :");
    scanf("%d",&a[i]);
    }
    mini=max=a[0];
    for(i=1;i < n;i++)
    {
    if(mini > a[i])
    mini=a[i];
    if(max < a[i])
    max=a[i];
    }
    printf("\nMinimum Number:%d",mini);
    printf("\nMaximum Number:%d",max);
    getch();
    }


  • Output:
    • How many numbers you want to print:3

    Enter Value of element:2
    Enter Value of element:1
    Enter Value of element:3

    Minimum Number:1
    Maximum Number:3

    4.Write a program to enter N elements and find how many are positive,negative,even and odd number.



  • C program:

    • #include<stdio.h>
    #include<conio.h>

    void main()
    {
    int a[100],i,n,positive=0,negative=0,even=0,odd=0;
    clrscr();
    printf("\nHow many numbers you want to print:");
    scanf("%d",&n);
    for(i=0;i < n;i++)
    {
    printf("\nEnter Value of element :");
    scanf("%d",&a[i]);
    }
    for(i=0;i < n;i++)
    {
    if(a[i] < 0)
    negative++;
    else
    positive++;
    if(a[i]%2==0)
    even++;
    else
    odd++;
    }
    printf("\nPositive :%d",positive);
    printf("\nNegative :%d",negative);
    printf("\nEven :%d",even);
    printf("\nOdd :%d",odd);
    getch();
    }


  • Output:
    • How many numbers you want to print:3

    Enter Value of element:-2
    Enter Value of element:1
    Enter Value of element:3

    Positive :2
    Negative :1
    Even :1
    Odd :2
